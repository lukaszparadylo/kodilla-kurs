CREATE TABLE ISSUESLISTS
(
    ID SERIAL PRIMARY KEY,
    NAME VARCHAR(100)
);

CREATE TABLE ISSUES
(
    ID  SERIAL PRIMARY KEY,
    ISSUESLIST_ID  BIGINT UNSIGNED NOT NULL,
    SUMMARY VARCHAR(100),
    BODY VARCHAR(1024),
    FOREIGN KEY (ISSUESLIST_ID) REFERENCES ISSUESLISTS(ID)
);
