SELECT * FROM USERS;

SELECT I.SUMMARY, IL.NAME
FROM ISSUES I JOIN  ISSUESLISTS IL ON IL.ID = I.ISSUESLIST_ID;

SELECT I.SUMMARY, U.FIRSTNAME, U.LASTNAME
FROM ISSUES I JOIN USERS U ON U.ID = I.ISSUESLIST_ID;

SELECT I.SUMMARY, U.FIRSTNAME, U.LASTNAME, COUNT(*) AS ISSUES_NUMBER
FROM ISSUES I JOIN USERS U ON U.ID = I.ISSUESLIST_ID
GROUP BY U.LASTNAME
ORDER BY U.LASTNAME;
